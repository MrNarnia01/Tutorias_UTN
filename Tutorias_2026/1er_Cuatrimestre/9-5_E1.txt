    Ejemplo de ordenamiento de vectores
    
*/

#include <iostream>
using namespace std;

int main(){
    
    int n=0;
    cin>>n;
    
    int v[n];
    for (int i = 0; i < n; i++) {
        v[i]=0;
    }
    
    for (int i = 0; i < n; i++) {
        cout<<"\nPosicion: "<<i+1<<endl;
        cin>>v[i];
    }
    
    for (int i = 0; i < n-1; i++) {
        for (int j = i+1; j < n; j++) {
            
            if(v[i]>v[j]){
                
                int aux=v[i];
                v[i]=v[j];
                v[j]=aux;
                
            }
            
        }
    }
    
    
    for (int i = 0; i < n; i++) {
        cout<<endl<<v[i]<<endl;
    }
    
    
    return 0;
}
